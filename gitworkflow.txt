//拉取远程仓库
git clone url
//创建新功能分支
git checkout -b my-feature
//修改代码增加功能后查看代码改变
git diff
//添加改变文件到暂存区
git add filename
//提交到local git
git commit filename
//提交到远程仓库
git push origin my-feature

//main分支有更新，拉取main
git pull origin main
//切换到功能分支并rebase，测试在main更新的基础上功能代码是否正常工作
git checkout my-feature
git rebase main
//测试完强制提交到远程仓库
git push -f origin my-feature
//提交pull request，请求main合并功能分支
new pull request
squash and merge

//本地删除功能分支
git branch -D my-feature
//拉取远程main
git pull origin main
